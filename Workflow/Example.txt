states:
    Draft
    Approval (approvers)
    ChangeNeeded (reason)
    Published

transitions:
    on "request approval from {approvers}"
        Draft -> Approval(approvers)
        ChangeNeeded -> Approval(approvers)

    on "approved by {approver}"
        Approval(approver) -> Published
        Approval(approvers) -> Approval(approvers - approver)

    on "request change {reason}"
        Approval(_) -> ChangeNeeded
